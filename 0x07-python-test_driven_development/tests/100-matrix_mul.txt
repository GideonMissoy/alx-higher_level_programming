import unittest
from parameterized import parameterized
from matrix_mul import matrix_mul

class TestMatrixMul(unittest.TestCase):

    def test_matrix_mul(self):
        self.assertEqual(matrix_mul([[1, 2], [3, 4]], [[1, 2], [3, 4]]), [[7, 10], [15, 22]])
        self.assertEqual(matrix_mul([[1, 2]], [[3, 4], [5, 6]]), [[13, 16]])
        self.assertEqual(matrix_mul([[1, 2, 3]], [[3, 4, 5], [5, 6, 8], [9, 10, 11]]), [[40, 46, 54]])
        self.assertEqual(matrix_mul([[2]], [[7]]), [[14]])
        self.assertRaises(TypeError, matrix_mul, 9, [[1, 2], [3, 4]])
        self.assertRaises(TypeError, matrix_mul, [[1, 2], [3, 4]], 9)
        self.assertRaises(TypeError, matrix_mul, 9, 9)
        self.assertRaises(TypeError, matrix_mul, [1, 2, 3, 4], [[1, 2], [3, 4]])
        self.assertRaises(TypeError, matrix_mul, [[1, 2], [3, 4]], [1, 2, 3, 4])
        self.assertRaises(TypeError, matrix_mul, [1, 2, 3, 4], [1, 2, 3, 4])
        self.assertRaises(ValueError, matrix_mul, [], [[1, 2], [3, 4]])
        self.assertRaises(ValueError, matrix_mul, [[]], [[1, 2], [3, 4]])
        self.assertRaises(ValueError, matrix_mul, [[1, 2], [3, 4]], [])
        self.assertRaises(ValueError, matrix_mul, [[1, 2], [3, 4]], [[]])
        self.assertRaises(ValueError, matrix_mul, [], [])
        self.assertRaises(ValueError, matrix_mul, [[]], [[]])
        self.assertRaises(TypeError, matrix_mul, [[1, 2], ["Betty", 4]], [[1, 2], [3, 4]])
        self.assertRaises(TypeError, matrix_mul, [[1, 2], [3, 4]], [[1, "Holberton"], [3, 4]])
        self.assertRaises(TypeError, matrix_mul, [[1, 2], [None, 4]], [[1, 2], [3, 4]])
        self.assertRaises(TypeError, matrix_mul, [[1, 2], [3]], [[1, 2], [3, 4]])
        self.assertRaises(TypeError, matrix_mul, [[1, 2], [3, 4]], [[1], [3, 4]])
        self.assertRaises(TypeError, matrix_mul, [[1, 2], [3]], [[1], [3, 4]])
        self.assertRaises(ValueError, matrix_mul, [[1, 2], [3, 4]], [[1, 2], [3, 4], [5, 6]])
        self.assertRaises(ValueError, matrix_mul, [[1, 2, 3], [3, 4, 5]], [[1, 2], [3, 4]])
